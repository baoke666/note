	$("input[name='btn']").click(function(){
		var imgFile = new FileReader();
		var img = document.getElementById('img');
		var info = img.files[0];
		imgFile.readAsDataURL(info);
		imgFile.onload=function(){
			var data = this.result;
			send(data);
		}
	});
	function send(img){
		var data = {};
		data.info = img;
		$.ajax({
			type : "post",
			dataType : "json",
			url : "base",
			data:data,
			success:function(msg){

			}
		});
	}
});
</script>
